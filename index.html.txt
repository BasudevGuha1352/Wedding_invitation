<script>
    const song = document.getElementById("weddingMusic");
    let isPlaying = false;
    let audioInitialized = false;

    // Unlocks audio context on first user interaction
    function initAudio() {
        if (!audioInitialized) {
            song.play().then(() => {
                if (!isPlaying) song.pause(); 
                audioInitialized = true;
            }).catch(() => {});
        }
    }

    function toggleMusic() {
        if (!isPlaying) {
            song.play();
            document.getElementById("musicLabel").innerText = "MUSIC ON üîä";
            isPlaying = true;
        } else {
            song.pause();
            document.getElementById("musicLabel").innerText = "MUSIC OFF üîá";
            isPlaying = false;
        }
    }

    function selectDay(dayId, btnId) {
        document.querySelectorAll('.day-btn').forEach(b => b.classList.remove('active'));
        document.querySelectorAll('.ceremony-list').forEach(l => l.classList.remove('active'));
        document.getElementById('detailsBox').style.display = 'none';
        document.getElementById(btnId).classList.add('active');
        document.getElementById(dayId).classList.add('active');
    }

    function showDet(name, time, venue, quote, map) {
        const box = document.getElementById('detailsBox');
        document.getElementById('detName').innerText = name;
        document.getElementById('detInfo').innerHTML = "<b>Time:</b> " + time + "<br><b>Venue:</b> " + venue + "<br><i>" + quote + "</i>";
        document.getElementById('mapLink').href = map;
        box.style.display = 'block';
    }
</script>
